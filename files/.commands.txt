#!/bin/bash

SHORT1=../data/short-reads1.fq.gz
SHORT2=../data/short-reads2.fq.gz
LONG_HIGH=../data/long_reads_high_depth.fastq.gz
LONG_LOW=../data/long_reads_low_depth.fastq.gz

## 1. run successive SPAdes assemblies for different k-mer length
unicycler -1 $SHORT1 -2 $SHORT2 --no_pilon --no_correct --kmers 31 -o out1_illumina/k31
unicycler -1 $SHORT1 -2 $SHORT2 --no_pilon --no_correct --kmers 47 -o out1_illumina/k47
unicycler -1 $SHORT1 -2 $SHORT2 --no_pilon --no_correct --kmers 63 -o out1_illumina/k63
unicycler -1 $SHORT1 -2 $SHORT2 --no_pilon --no_correct --kmers 89 -o out1_illumina/k89

## 2. run Unicycler pipeline (Illumina-only)
unicycler -1 $SHORT1 -2 $SHORT2 --no_correct -o out1_illumina/unicycler

## 3. run Unicycler pipeline (Hybrid, low coverage)
unicycler -1 $SHORT1 -2 $SHORT2 -l $LONG_LOW --no_correct -o out2_hybrid/low-cov

## 4. run Unicycler pipeline (Hybrid, high coverage)
unicycler -1 $SHORT1 -2 $SHORT2 -l $LONG_HIGH --no_correct -o out2_hybrid/high-cov
